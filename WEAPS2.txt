actor ShockRifle2000 : GoHWeapon 6084
{
  Inventory.PickupMessage "T-2000 Shock Rifle"
  Inventory.PickupSound "weapons/getweapon07"
  obituary "%o had no chance to survive %k's 2000 Shock Rifle."
  Weapon.AmmoGive 0
  Weapon.AmmoType "Cell"
  Weapon.Ammouse 2
  Weapon.Kickback 20
  +NOALERT
  states
  {
  Spawn:
    SHOC A -1
    LOOP
  Ready:
    SHCK A 0 A_JumpIfInventory("firemodechecker2",1,"Ready2")
  Ready1:
    SHCK ABCDEDCB 1 A_WeaponReady(WRF_ALLOWUSER1)
    loop
  Ready2:
    SHK2 ABCDEDCB 1 A_WeaponReady(WRF_ALLOWUSER1)
    loop
  User1: 
    TNT1 A 0 A_SpawnItemEx("LimitShockRifle2000", 70, 0, -10, 15, 0, 5, 0, SXF_SETMASTER)
    TNT1 A 0 A_TakeInventory("Token",1)
    TNT1 A 0 A_TakeInventory("ShockRifle2000",1)
    stop
  Deselect:
    SHCK E 1 A_Lower
	NULL AA 0 A_Lower
    loop
  Select:
    SHCK E 1 A_Raise
	NULL AA 0 A_RAISE
    loop
  Fire:
    TNT1 A 0 A_JumpIfNoAmmo("Deselect")
    SHCK E 0 A_JumpIfInventory("firemodechecker2",1,"Fire2")	
   Fire1:
      SHCK F 1 Bright
      SHCK G 0 A_Light1
      SHCK G 0 A_PlayWeaponSound("weapons/shock")
      SHCK GH 1 Bright A_FireProjectile("Lightning2",0.5,1,6,1)
      SHCK I 0 A_Light2
      SHCK I 0 ACS_Execute(287,0,30,random(2,4),0)
      SHCK I 1 Bright A_FireProjectile("Lightning2",0.5,0,6,1)
      SHCK A 0 A_Light0
      SHCK I 0 ACS_Execute(287,0,-30,random(-2,-4),0)
      SHCK A 1 A_ReFire
      Goto Ready
   Fire2:
      SHK2 F 0 A_Light1
      SHK2 F 0 A_PlayWeaponSound("weapons/microwave")
      SHCK I 0 ACS_Execute(287,0,30,random(2,4),0)
      SHK2 F 1 A_RailAttack(53,5,1,"FF 00 00",none,1,1)
      SHK2 G 0 A_Light0
      SHK2 G 1 Bright
      SHCK I 0 ACS_Execute(287,0,-30,random(-2,-4),0)
      SHK2 HI 1
      SHK2 A 2
      SHK2 A 1 A_ReFire
      Goto Ready	
  AltFire: 
    SHCK A 0 A_Playsound("weapons/prototyperifleswitch") 
    SHCK A 0 A_GiveInventory("firemodechecker2",1) 
    SHCK A 0 A_JumpIfInventory("firemodechecker2",2,"AltFire2") 
    SHCK I 5
    SHK2 A 5
    goto Ready
  AltFire2:
    SHK2 I 5 A_TakeInventory("firemodechecker2",2) 
    SHCK A 5
    goto Ready
  }
}

actor PlasmaAutoRifle : GoHWeapon
{
  Weapon.SelectionOrder 100
  Weapon.AmmoUse 1
  Weapon.AmmoGive 0
  Weapon.AmmoType "Cell"
  Inventory.PickupMessage "You got the plasma autorifle!"
  inventory.pickupsound "Pickup/Gun"
  Weapon.upsound "Auto/Up"
  States
  {
  Ready:
    AUTO A 1 A_WeaponReady(WRF_ALLOWUSER1)
    Loop
  User1: 
	TNT1 A 0 A_SpawnItemEx("LimitPlasmaAutoRifle", 70, 0, -10, 15, 0, 5, 0, SXF_SETMASTER)
	TNT1 A 0 A_TakeInventory("Token",1)
	TNT1 A 0 A_TakeInventory("PlasmaAutoRifle",1)
	stop
  Deselect:
    AUTO A 1 A_Lower
	NULL AA 0 A_Lower
    Loop
  Select:
    AUTO A 1 A_Raise
	NULL AA 0 A_Raise
    Loop
  Fire:
    TNT1 A 0 A_JumpIfNoAmmo("Deselect")
	AUTO A 1 A_TakeInventory ("AutoCounter",30)
  Hold:
    AUTO B 1 A_FireProjectile (AutoPlasma, 0, 1, 8, 3)
	AUTO C 1
    AUTO B 0 A_FireProjectile (PlasmaSmoke,Random(2,-2),0,8,3)
    AUTO A 0 ACS_Execute(287,0,random(-26,26),random(-6,6),0)
    AUTO B 0 A_PlayWeaponSound ("Auto/Fire")
    AUTO A 2 A_GiveInventory ("AutoCounter",1)
    AUTO A 0 A_JumpIfInventory ("AutoCounter",30,"Overheat")
    AUTO A 0 A_ReFire("Hold")
    AUTO A 12 A_TakeInventory ("AutoCounter",30)
    Goto Ready
  AltFire:
    TNT1 A 0 A_JumpIfNoAmmo("Deselect")
    AUTO A 5 A_PlaySound("Auto/Charge", CHAN_WEAPON, 1.0, 0)
    AUTO A 0 A_ReFire("AltCharge1")
    Goto Ready
  AltCharge2:
    AUTO AAAA 10
    AUTO A 0 A_ReFire("AltCharge2")
    Goto Ready
  AltCharge3:
    AUTO AAAA 10
    AUTO A 0 A_ReFire("AltCharge3")
    Goto Ready
  AltCharge4:
    AUTO AAAA 10
    AUTO A 0 A_ReFire("AltHold")
    Goto Ready
  AltHold:
    AUTO B 5 A_FireProjectile (BigAutoPlasma, 0, 3, 8, 3)
	AUTO C 5
    AUTO BBB 0 A_FireProjectile (PlasmaSmoke,Random(2,-2),0,8,3)
    AUTO A 0 ACS_Execute(287,0,random(-26,26),random(-6,6),0)
    AUTO B 0 A_PlayWeaponSound ("Auto/Fire")
    AUTO A 10 A_GiveInventory ("AutoCounter",10)
    AUTO A 0 A_JumpIfInventory ("AutoCounter",30,"Overheat")
    Goto Ready
  OverHeat:
    AUTO D 2 A_FireProjectile (PlasmaSmoke2,Random(2,-2),0,8,3)
    AUTO B 0 A_PlayWeaponSound ("Auto/Overheat")
    AUTO DDDDDDDDDDDDDDDDDDDDDDDDDD 2 A_FireProjectile (PlasmaSmoke2,Random(2,-2),0,8,3)
    AUTO A 16 A_TakeInventory ("AutoCounter",30)
    goto Ready
  Flash:
    TNT1 A 4 bright A_Light1
    TNT1 A 5 bright A_Light2
    TNT1 A 0 bright A_Light0
    stop
  Spawn:
    AUTP A -1
    Stop
  }
}
